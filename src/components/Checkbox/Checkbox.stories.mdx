# Checkbox

A Checkbox component is an enhanced version of the default HTML `<input>` `checkbox`. Checkbox uses the `Choice` component as a smart wrapper.

<Preview>
  <Story name="default">
    <ChoiceGroup
      multiSelect={boolean('multiSelect', true)}
      multiSelectLimit={2}
      value={select(
        'Value from props',
        {
          zero: [],
          one: ['hansel'],
          two: ['hansel', 'mugatu'],
          three: ['hansel', 'mugatu', 'derek'],
        },
        ['hansel', 'mugatu']
      )}
    >
      <Checkbox label="Derek" value="derek" />
      <Checkbox label="Hansel" value="hansel" />
      <Checkbox label="Mugatu" value="mugatu" />
      <Checkbox label="John" value="john" />
      <Checkbox label="Paul" value="paul" />
    </ChoiceGroup>
  </Story>
</Preview>
